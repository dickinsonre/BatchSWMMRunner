      // Modify the .inp content with the new parameter value
      const modified = modifyParameter(
        baseInpContent, 
        config.parameterName, 
        value
      );
      
      // Create a virtual file for upload
      const blob = new Blob([modified], { type: 'text/plain' });
      const fileName = `sweep_${config.parameterName}_${value}.inp`;
      modifiedFiles.push(new File([blob], fileName));

  // SCS Type II distribution (normalized cumulative)
  const SCS_TYPE_II = [
    [0.0, 0.000], [0.1, 0.022], [0.2, 0.048],
    [0.3, 0.080], [0.35, 0.120], [0.4, 0.181],
    [0.42, 0.235], [0.44, 0.332], [0.46, 0.500],
    [0.48, 0.668], [0.50, 0.765], [0.52, 0.819],
    [0.55, 0.880], [0.6, 0.920], [0.7, 0.952],
    [0.8, 0.973], [0.9, 0.989], [1.0, 1.000]
  ];
  
  function generateStormTimeseries(
    totalDepth: number, 
    durationHours: number
  ): string {
    let ts = '[TIMESERIES]\n;;Name Time Value\n';
    
    for (let i = 0; i < SCS_TYPE_II.length; i++) {
      const [tFrac, pFrac] = SCS_TYPE_II[i];
      const prevPFrac = i > 0 ? SCS_TYPE_II[i-1][1] : 0;
      const hours = tFrac * durationHours;
      const incrementalDepth = (pFrac - prevPFrac) * totalDepth;
      
      ts += `DesignStorm  ${hours.toFixed(2)}  ${incrementalDepth.toFixed(4)}\n`;
    }
    
    return ts;
  }
